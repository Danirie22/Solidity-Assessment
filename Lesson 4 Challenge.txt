pragma solidity ^0.8.7;

contract Operators{
    function add (int number1, int number2) external pure returns(int){
        return number1 + number2;
    }
    function subtract (int number1, int number2) external pure returns(int){
        return number1 - number2;
    }
    function multiply (int number1, int number2) external pure returns(int){
        return number1 * number2;
    }
    function divide (int number1, int number2) external pure returns(int){
        return number1 / number2;
    }
}
